```javascript
function spawnMonkey() {
  const matureTrees = plants.filter(p => p.type === 'tree' && p.done);
  if (matureTrees.length < 10) return;

  const dir = Math.random() < 0.5 ? 1 : -1;
  const startX = dir === 1 ? -5 : W + 5;

  // Find first reachable tree in travel direction
  const sorted = matureTrees
    .filter(t => dir === 1 ? t.x > 0 : t.x < W)
    .sort((a, b) => dir === 1 ? a.x - b.x : b.x - a.x);

  if (sorted.length < 3) return; // not enough trees to traverse

  const firstTree = sorted[0];
  const gripX = firstTree.x;
  const gripY = firstTree.ground - firstTree.height;

  animals.push({
    type: 'monkey',
    x: startX,
    y: gripY,
    dir,
    state: 'swinging',
    gripX: startX,
    gripY: gripY,
    nextGripX: gripX,
    nextGripY: gripY,
    swingFrame: 0,
    swingTotal: 10,
    pauseTimer: 0,
    life: 9999,
  });
}
```

```javascript
function findNextGrip(monkey) {
  const matureTrees = plants.filter(p => p.type === 'tree' && p.done);
  const dir = monkey.dir;
  const cx = monkey.nextGripX;

  // Find trees ahead within 60px, then try 100px
  for (const range of [60, 100]) {
    const candidates = matureTrees.filter(t => {
      const dx = (t.x - cx) * dir;
      return dx > 5 && dx <= range;
    });
    if (candidates.length > 0) {
      candidates.sort((a, b) => (a.x - cx) * dir - (b.x - cx) * dir);
      const tree = candidates[0];
      return { x: tree.x, y: tree.ground - tree.height };
    }
  }
  return null; // no tree reachable
}
```

```javascript
function drawMonkey(m) {
  const body = [140, 90, 40];
  const tail = [110, 70, 30];

  if (m.state === 'swinging') {
    // Body: 2px horizontal
    setOverlay(m.x, m.y, body);
    setOverlay(m.x + m.dir, m.y, body);
    // Arm reaching up toward grip direction
    const armDx = -m.dir;
    setOverlay(m.x + armDx, m.y - 1, body);
    // Tail: 2px trailing behind and curving down
    setOverlay(m.x - m.dir, m.y + 1, tail);
    setOverlay(m.x - m.dir * 2, m.y + 2, tail);
  } else {
    // Gripping: vertical hanging pose
    setOverlay(m.x, m.y, body);
    setOverlay(m.x, m.y + 1, body);
    // Arm above at grip
    setOverlay(m.x, m.y - 1, body);
    // Tail curling below
    setOverlay(m.x + 1, m.y + 2, tail);
    setOverlay(m.x + 1, m.y + 3, tail);
  }
}
```
